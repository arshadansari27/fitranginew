As ubuntu
=========
sudo apt-get update
sudo apt-get install python-setuptools mongodb libjpeg-dev libtiff-dev libpng-dev zlib1g-dev libffi-dev htop python-dev git libssl-dev nginx
mkdir buildenv
cd buildenv
git clone git@github.com:arshadansari27/fitranginew.git fitrangi
cd fitrangi
sudo easy_install pip
sudo pip install gunicorn supervisor
sudo pip install -r requirements.new.txt

cd /etc/nginx/conf.d
sudo ln -s /home/ubuntu/buildenv/fitrangi/blacklist.conf blacklist.conf

cd /etc/nginx/sites-enabled
sudo ln -s /home/ubuntu/buildenv/fitrangi/fitrangi.nginx.conf fitrangi.nginx.conf
